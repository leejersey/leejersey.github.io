# 第五章 分布式锁

> 本章目标：从最简单的 SETNX 锁到生产级分布式锁，理解锁过期、误删、续约问题，用 contextmanager 封装优雅的锁接口。

## 5.1 从 SETNX 到生产级分布式锁

### 为什么需要分布式锁？

单进程应用可以用 `threading.Lock()` 解决并发问题。但当应用部署了多个实例（多台服务器），线程锁就失效了——不同服务器上的进程无法共享内存中的锁。

```
❌ 线程锁：只在一个进程内有效

  服务器 A                服务器 B
  ┌──────────┐           ┌──────────┐
  │ Lock ✅  │           │ Lock ✅  │  ← 两边都拿到了"锁"
  │ 扣库存    │           │ 扣库存    │  ← 同时扣，超卖了！
  └──────────┘           └──────────┘

✅ 分布式锁：所有服务器共享一把锁

  服务器 A                服务器 B
  ┌──────────┐           ┌──────────┐
  │ Redis ✅ │           │ Redis ❌ │  ← B 拿不到锁，等待
  │ 扣库存    │           │ 等待...   │
  └──────────┘           └──────────┘
       ↕                      ↕
  ┌──────────────────────────────┐
  │       Redis（共享锁存储）      │
  └──────────────────────────────┘
```

### 最简单的锁：SETNX

```python
import redis
import uuid

r = redis.Redis(host="127.0.0.1", port=6379, db=0, decode_responses=True)

def acquire_lock(lock_name: str, timeout: int = 10) -> str | None:
    """获取锁，返回锁的唯一标识（用于释放时验证）"""
    lock_key = f"lock:{lock_name}"
    lock_value = str(uuid.uuid4())  # 每次生成唯一 ID

    # SET key value NX EX timeout
    if r.set(lock_key, lock_value, nx=True, ex=timeout):
        return lock_value  # 获取成功
    return None  # 获取失败

def release_lock(lock_name: str, lock_value: str) -> bool:
    """释放锁"""
    lock_key = f"lock:{lock_name}"
    current = r.get(lock_key)
    if current == lock_value:
        r.delete(lock_key)
        return True
    return False
```

### 等等，这个释放有 Bug！

上面的 `release_lock` 有一个**竞态条件**：

```
时刻 1: 线程 A 执行 GET lock_key → 值匹配 ✅
时刻 2: 锁恰好过期了，被自动删除
时刻 3: 线程 B 获取到了新锁
时刻 4: 线程 A 执行 DELETE lock_key → 删掉了线程 B 的锁！💥
```

GET 和 DELETE 不是原子的。解决方案：**用 Lua 脚本原子释放**。

### 生产级实现

```python
import redis
import uuid
import time

r = redis.Redis(host="127.0.0.1", port=6379, db=0, decode_responses=True)

# 原子释放锁的 Lua 脚本
_RELEASE_SCRIPT = r.register_script("""
if redis.call('GET', KEYS[1]) == ARGV[1] then
    return redis.call('DEL', KEYS[1])
else
    return 0
end
""")


class DistributedLock:
    """生产级分布式锁"""

    def __init__(
        self,
        name: str,
        timeout: int = 10,
        retry_times: int = 3,
        retry_delay: float = 0.2,
        redis_client: redis.Redis = None,
    ):
        self.name = name
        self.key = f"lock:{name}"
        self.timeout = timeout
        self.retry_times = retry_times
        self.retry_delay = retry_delay
        self.r = redis_client or r
        self._value: str | None = None

    def acquire(self) -> bool:
        """获取锁（带重试）"""
        self._value = str(uuid.uuid4())

        for attempt in range(self.retry_times):
            if self.r.set(self.key, self._value, nx=True, ex=self.timeout):
                return True

            if attempt < self.retry_times - 1:
                time.sleep(self.retry_delay)

        self._value = None
        return False

    def release(self) -> bool:
        """释放锁（原子操作）"""
        if self._value is None:
            return False
        result = _RELEASE_SCRIPT(keys=[self.key], args=[self._value])
        self._value = None
        return result == 1

    @property
    def is_locked(self) -> bool:
        """检查锁是否被持有"""
        return self.r.exists(self.key) == 1


# 使用示例
lock = DistributedLock("order:1001", timeout=10)
if lock.acquire():
    try:
        print("获取到锁，执行业务逻辑...")
        # 扣库存、创建订单等
    finally:
        lock.release()
else:
    print("获取锁失败，稍后重试")
```

### 关键设计要点

| 要点 | 说明 |
|------|------|
| **唯一 value** | 用 UUID，确保只有加锁者能解锁 |
| **原子释放** | Lua 脚本保证 GET+DELETE 原子执行 |
| **超时自动释放** | 防止持有者崩溃导致死锁 |
| **重试机制** | 获取失败不立即放弃，间隔重试 |

## 5.2 锁续约与看门狗机制

### 问题：业务执行时间 > 锁超时时间

```
设置锁超时 = 10 秒
业务实际执行 = 15 秒

时刻 0s:   线程 A 获取锁
时刻 10s:  锁自动过期 💥
时刻 10.1s: 线程 B 获取到锁
时刻 15s:  线程 A 还在执行 → 两个线程同时操作！
```

设太短会"锁提前释放"，设太长会"锁持有太久影响并发"。

### 解决方案：看门狗（Watchdog）自动续约

```
线程 A 获取锁（timeout=10s）
  │
  ├── 启动看门狗线程
  │   └── 每 timeout/3 秒（~3.3s）检查一次
  │       └── 如果锁还是自己的 → 续约（重置为 10s）
  │       └── 如果锁不是自己的 → 停止
  │
  ├── 执行业务逻辑（可能很久）
  │
  └── 业务完成 → 释放锁 → 停止看门狗
```

### Python 实现

```python
import threading


class DistributedLockWithWatchdog:
    """带看门狗续约的分布式锁"""

    def __init__(
        self,
        name: str,
        timeout: int = 10,
        retry_times: int = 3,
        retry_delay: float = 0.2,
        redis_client: redis.Redis = None,
    ):
        self.name = name
        self.key = f"lock:{name}"
        self.timeout = timeout
        self.retry_times = retry_times
        self.retry_delay = retry_delay
        self.r = redis_client or r
        self._value: str | None = None
        self._watchdog: threading.Timer | None = None
        self._watchdog_running = False

    def acquire(self) -> bool:
        """获取锁并启动看门狗"""
        self._value = str(uuid.uuid4())

        for attempt in range(self.retry_times):
            if self.r.set(self.key, self._value, nx=True, ex=self.timeout):
                self._start_watchdog()
                return True
            if attempt < self.retry_times - 1:
                time.sleep(self.retry_delay)

        self._value = None
        return False

    def release(self) -> bool:
        """释放锁并停止看门狗"""
        self._stop_watchdog()
        if self._value is None:
            return False
        result = _RELEASE_SCRIPT(keys=[self.key], args=[self._value])
        self._value = None
        return result == 1

    def _start_watchdog(self):
        """启动看门狗：定期续约"""
        self._watchdog_running = True
        self._schedule_renewal()

    def _stop_watchdog(self):
        """停止看门狗"""
        self._watchdog_running = False
        if self._watchdog:
            self._watchdog.cancel()
            self._watchdog = None

    def _schedule_renewal(self):
        """定时续约"""
        if not self._watchdog_running:
            return
        interval = self.timeout / 3  # 每 1/3 超时时间续约一次
        self._watchdog = threading.Timer(interval, self._renew)
        self._watchdog.daemon = True  # 主线程退出时自动结束
        self._watchdog.start()

    def _renew(self):
        """执行续约"""
        if not self._watchdog_running or self._value is None:
            return
        # 只有值匹配才续约（确认还是自己的锁）
        current = self.r.get(self.key)
        if current == self._value:
            self.r.expire(self.key, self.timeout)
            self._schedule_renewal()  # 安排下一次
        else:
            self._watchdog_running = False


# 使用
lock = DistributedLockWithWatchdog("long_task", timeout=10)
if lock.acquire():
    try:
        print("开始执行长任务...")
        time.sleep(25)  # 模拟 25 秒的长任务
        print("长任务完成！")
        # 看门狗在这 25 秒内自动续约了多次，锁不会过期
    finally:
        lock.release()
```

> 💡 **Java 生态中的 Redisson** 就是这么做的——内置看门狗自动续约。Python 中需要自己实现，上面就是核心逻辑。

## 5.3 用 contextmanager 封装锁

### 目标：像 threading.Lock 一样优雅

```python
# threading.Lock 的用法
with threading.Lock():
    do_something()

# 我们要实现的效果
with distributed_lock("order:1001"):
    do_something()
```

### 方式一：给类加 `__enter__` / `__exit__`

```python
class DistributedLock:
    # ... 之前的代码 ...

    def __enter__(self):
        if not self.acquire():
            raise TimeoutError(f"获取锁 '{self.name}' 超时")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.release()
        return False  # 不吞异常

# 使用
with DistributedLock("order:1001", timeout=10) as lock:
    print("在锁保护内执行...")
    # 自动释放，即使抛异常
```

### 方式二：用 contextmanager 工厂函数

```python
from contextlib import contextmanager


@contextmanager
def distributed_lock(
    name: str,
    timeout: int = 10,
    retry_times: int = 3,
    redis_client: redis.Redis = None,
):
    """分布式锁上下文管理器"""
    client = redis_client or r
    lock_key = f"lock:{name}"
    lock_value = str(uuid.uuid4())

    # 获取锁
    acquired = False
    for _ in range(retry_times):
        if client.set(lock_key, lock_value, nx=True, ex=timeout):
            acquired = True
            break
        time.sleep(0.2)

    if not acquired:
        raise TimeoutError(f"获取锁 '{name}' 超时")

    try:
        yield lock_value  # 把锁的 value 交给调用者（通常用不到）
    finally:
        # 原子释放
        _RELEASE_SCRIPT(keys=[lock_key], args=[lock_value], client=client)


# 使用：简洁优雅
with distributed_lock("order:1001"):
    print("扣库存...")
    print("创建订单...")

# 可以捕获超时
try:
    with distributed_lock("hot_resource", timeout=5, retry_times=10):
        process_hot_resource()
except TimeoutError:
    print("资源繁忙，请稍后重试")
```

### 异步版本

```python
from contextlib import asynccontextmanager
import redis.asyncio as aioredis


@asynccontextmanager
async def async_distributed_lock(
    name: str,
    timeout: int = 10,
    retry_times: int = 3,
    redis_client: aioredis.Redis = None,
):
    """异步分布式锁"""
    client = redis_client or async_r
    lock_key = f"lock:{name}"
    lock_value = str(uuid.uuid4())

    acquired = False
    for _ in range(retry_times):
        if await client.set(lock_key, lock_value, nx=True, ex=timeout):
            acquired = True
            break
        await asyncio.sleep(0.2)

    if not acquired:
        raise TimeoutError(f"获取锁 '{name}' 超时")

    try:
        yield lock_value
    finally:
        await client.eval(
            "if redis.call('GET',KEYS[1])==ARGV[1] then return redis.call('DEL',KEYS[1]) else return 0 end",
            1, lock_key, lock_value,
        )


# 在 FastAPI 中使用
from fastapi import FastAPI

app = FastAPI()

@app.post("/orders/{order_id}/pay")
async def pay_order(order_id: int):
    async with async_distributed_lock(f"order:{order_id}", timeout=10):
        # 这里保证同一订单只有一个请求在处理
        await process_payment(order_id)
        return {"status": "success"}
```

### 三种封装方式对比

| 方式 | 代码量 | 灵活性 | 适合 |
|------|--------|--------|------|
| **手动 acquire/release** | 多 | 最高（可单独控制） | 复杂场景 |
| **类 + with** | 中 | 高（可访问锁对象） | 需要锁状态信息 |
| **contextmanager** | 最少 | 中（用完即走） | 大部分场景 ✅ |

## 本章小结

| 知识点 | 要点 |
|--------|------|
| 基础锁 | `SET key value NX EX timeout`，value 用 UUID |
| 原子释放 | Lua 脚本：GET + 比对 + DEL 一步完成 |
| 看门狗 | 每 timeout/3 自动续约，防止长任务锁过期 |
| contextmanager | `with distributed_lock("name"):` 最优雅 |
| 异步版 | `async with async_distributed_lock("name"):` 适配 FastAPI |
