## 4.1 Cache-Aside 模式：读写流程完整实现

### 什么是 Cache-Aside？

Cache-Aside（旁路缓存）是最常用的缓存模式——应用代码**自己管理** Redis 和数据库之间的数据同步：

```
读流程：
┌────────┐    ① 查 Redis     ┌────────┐
│  应用   │ ──────────────→  │  Redis  │
│        │ ←── 命中？返回 ── │        │
│        │                   └────────┘
│        │    ② 未命中，查 DB  ┌────────┐
│        │ ──────────────→  │  MySQL  │
│        │ ←── 返回数据 ──── │        │
│        │                   └────────┘
│        │    ③ 回写 Redis
│        │ ──────────────→  Redis
└────────┘

写流程：
┌────────┐    ① 更新 DB      ┌────────┐
│  应用   │ ──────────────→  │  MySQL  │
│        │                   └────────┘
│        │    ② 删除缓存      ┌────────┐
│        │ ──────────────→  │  Redis  │
└────────┘                   └────────┘
```

### Python 完整实现

```python
import json
import redis

r = redis.Redis(host="127.0.0.1", port=6379, db=0, decode_responses=True)


def get_user(user_id: int) -> dict | None:
    """Cache-Aside 读流程"""
    cache_key = f"cache:user:{user_id}"

    # ① 查缓存
    cached = r.get(cache_key)
    if cached is not None:
        print(f"[HIT] 缓存命中: {cache_key}")
        return json.loads(cached)

    # ② 缓存未命中，查数据库
    print(f"[MISS] 缓存未命中，查询数据库...")
    user = query_user_from_db(user_id)  # 你的数据库查询函数

    if user is None:
        return None

    # ③ 回写缓存（设过期时间）
    r.set(cache_key, json.dumps(user, ensure_ascii=False), ex=3600)
    return user


def update_user(user_id: int, data: dict):
    """Cache-Aside 写流程"""
    cache_key = f"cache:user:{user_id}"

    # ① 先更新数据库
    update_user_in_db(user_id, data)  # 你的数据库更新函数

    # ② 再删除缓存（而不是更新缓存）
    r.delete(cache_key)
    print(f"[INVALIDATE] 已删除缓存: {cache_key}")


# 模拟数据库操作
def query_user_from_db(user_id: int) -> dict | None:
    """模拟数据库查询"""
    fake_db = {
        1001: {"id": 1001, "name": "张三", "age": 28},
        1002: {"id": 1002, "name": "李四", "age": 32},
    }
    return fake_db.get(user_id)

def update_user_in_db(user_id: int, data: dict):
    """模拟数据库更新"""
    print(f"[DB] 已更新用户 {user_id}: {data}")
```

### 为什么是"删缓存"而不是"更新缓存"？

```
❌ 更新缓存的问题（并发场景）：
  线程 A: 更新 DB (age=29) → 更新缓存 (age=29)
  线程 B: 更新 DB (age=30) → 更新缓存 (age=30)

  如果线程 B 先更新了缓存，线程 A 后更新缓存：
  DB 里是 age=30，缓存里是 age=29 → 不一致！

✅ 删缓存的好处：
  线程 A: 更新 DB (age=29) → 删缓存
  线程 B: 更新 DB (age=30) → 删缓存

  下次读取时重新从 DB 加载 → 永远是最新值
```

> 📌 **Cache-Aside 口诀**：读时回填，写时删除。

## 4.2 缓存三大问题：穿透、雪崩、击穿的 Python 解法

### 问题一：缓存穿透

**现象**：查询一个**根本不存在**的数据，缓存永远 MISS，每次都打到数据库。

```
恶意请求 user_id = -1（不存在的用户）
  → Redis MISS → 查 DB → DB 也没有 → 不缓存
  → Redis MISS → 查 DB → DB 也没有 → 不缓存
  → 无限循环打 DB...
```

**解法 1：缓存空值**

```python
def get_user_safe(user_id: int) -> dict | None:
    """防穿透：缓存空值"""
    cache_key = f"cache:user:{user_id}"

    cached = r.get(cache_key)
    if cached is not None:
        if cached == "NULL":  # 空值标记
            return None
        return json.loads(cached)

    user = query_user_from_db(user_id)

    if user is None:
        # 数据不存在也缓存，但 TTL 短一些
        r.set(cache_key, "NULL", ex=300)  # 5 分钟
        return None

    r.set(cache_key, json.dumps(user, ensure_ascii=False), ex=3600)
    return user
```

**解法 2：布隆过滤器**

在查缓存之前，先用布隆过滤器判断 key 是否"可能存在"：

```python
# pip install pybloom-live
from pybloom_live import BloomFilter

# 初始化时把所有合法 ID 加入布隆过滤器
bf = BloomFilter(capacity=1000000, error_rate=0.001)
for uid in get_all_user_ids_from_db():
    bf.add(uid)

def get_user_bloom(user_id: int) -> dict | None:
    """防穿透：布隆过滤器"""
    if user_id not in bf:
        return None  # 一定不存在，直接返回

    # 可能存在，走正常缓存流程
    return get_user_safe(user_id)
```

### 问题二：缓存雪崩

**现象**：大量缓存**同时过期**，请求瞬间全部打到数据库。

```
时刻 T：1000 个缓存同时到期
  → 1000 个请求同时 MISS
  → 1000 个请求同时查 DB
  → DB 瞬间过载甚至挂掉
```

**解法：随机过期时间**

```python
import random

def set_cache_with_jitter(key: str, value: str, base_ttl: int = 3600):
    """带随机抖动的缓存写入"""
    # 在基础 TTL 上加 ±10% 的随机偏移
    jitter = random.randint(-base_ttl // 10, base_ttl // 10)
    ttl = base_ttl + jitter
    r.set(key, value, ex=ttl)

# 同一批缓存的过期时间会分散在 3240s ~ 3960s 之间
# 避免同时过期
```

### 问题三：缓存击穿

**现象**：某个**热点 key** 过期的瞬间，大量请求同时打到数据库。

```
热点 key "hot:article:1" 过期
  → 100 个并发请求同时 MISS
  → 100 个请求同时查 DB 并回写缓存
  → DB 承受 100 倍压力，且 99 次回写是浪费
```

**解法：互斥锁（只让一个请求回源）**

```python
import time

def get_with_mutex(key: str, db_query_func, ttl: int = 3600):
    """防击穿：互斥锁回源"""
    cached = r.get(key)
    if cached is not None:
        return json.loads(cached) if cached != "NULL" else None

    # 尝试获取互斥锁
    lock_key = f"lock:{key}"
    if r.set(lock_key, "1", nx=True, ex=10):  # 锁 10 秒自动释放
        try:
            # 获取到锁，查数据库
            data = db_query_func()
            if data is not None:
                r.set(key, json.dumps(data, ensure_ascii=False), ex=ttl)
            else:
                r.set(key, "NULL", ex=300)
            return data
        finally:
            r.delete(lock_key)  # 释放锁
    else:
        # 没拿到锁，说明别人正在回源，等一下再读缓存
        time.sleep(0.1)
        cached = r.get(key)
        if cached and cached != "NULL":
            return json.loads(cached)
        return None  # 极端情况兜底
```

### 三大问题速查

| 问题 | 原因 | 解法 |
|------|------|------|
| **穿透** | 查不存在的数据 | 缓存空值 / 布隆过滤器 |
| **雪崩** | 大量 key 同时过期 | TTL 加随机抖动 |
| **击穿** | 热点 key 过期瞬间 | 互斥锁回源 |

## 4.3 用装饰器封装缓存逻辑

### 问题：到处写缓存样板代码

```python
# 每个函数都要写一遍：查缓存 → 未命中查 DB → 回写
def get_user(user_id):
    cached = r.get(f"cache:user:{user_id}")
    if cached: return json.loads(cached)
    user = query_db(user_id)
    r.set(f"cache:user:{user_id}", json.dumps(user), ex=3600)
    return user

def get_article(article_id):
    cached = r.get(f"cache:article:{article_id}")
    if cached: return json.loads(cached)
    article = query_db(article_id)
    r.set(f"cache:article:{article_id}", json.dumps(article), ex=3600)
    return article

# 几乎一样的代码重复了 N 遍...
```

### 解决方案：缓存装饰器

```python
import json
import functools
import hashlib
from typing import Callable


def redis_cache(
    prefix: str,
    ttl: int = 3600,
    null_ttl: int = 300,
    redis_client: redis.Redis = None,
):
    """
    Redis 缓存装饰器

    Args:
        prefix: 缓存 key 前缀，如 "cache:user"
        ttl: 正常数据的过期时间（秒）
        null_ttl: 空值的过期时间（秒），防穿透
        redis_client: Redis 连接实例
    """
    def decorator(func: Callable):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            client = redis_client or r  # 默认用全局 r

            # 生成缓存 key
            key_parts = [str(a) for a in args] + [f"{k}={v}" for k, v in sorted(kwargs.items())]
            key_suffix = ":".join(key_parts) if key_parts else "no_args"
            cache_key = f"{prefix}:{key_suffix}"

            # 查缓存
            cached = client.get(cache_key)
            if cached is not None:
                if cached == "__NULL__":
                    return None
                return json.loads(cached)

            # 未命中，执行原函数
            result = func(*args, **kwargs)

            # 回写缓存
            if result is not None:
                client.set(cache_key, json.dumps(result, ensure_ascii=False, default=str), ex=ttl)
            else:
                client.set(cache_key, "__NULL__", ex=null_ttl)

            return result
        return wrapper
    return decorator
```

### 使用装饰器

```python
@redis_cache(prefix="cache:user", ttl=3600)
def get_user(user_id: int) -> dict | None:
    """查询用户（自动缓存）"""
    print(f"[DB] 查询用户 {user_id}")
    return query_user_from_db(user_id)

@redis_cache(prefix="cache:article", ttl=1800)
def get_article(article_id: int) -> dict | None:
    """查询文章（自动缓存）"""
    print(f"[DB] 查询文章 {article_id}")
    return {"id": article_id, "title": "示例文章"}

# 第一次调用：查 DB + 写缓存
user = get_user(1001)     # 打印 [DB] 查询用户 1001

# 第二次调用：直接走缓存
user = get_user(1001)     # 不打印（缓存命中）

# 不同参数生成不同的 key
user = get_user(1002)     # 打印 [DB] 查询用户 1002
```

### 异步版本

```python
import redis.asyncio as aioredis

def async_redis_cache(prefix: str, ttl: int = 3600, null_ttl: int = 300):
    """异步版缓存装饰器"""
    def decorator(func: Callable):
        @functools.wraps(func)
        async def wrapper(*args, _redis: aioredis.Redis = None, **kwargs):
            client = _redis or async_r  # 全局异步 Redis 实例

            key_parts = [str(a) for a in args] + [f"{k}={v}" for k, v in sorted(kwargs.items())]
            cache_key = f"{prefix}:{':'.join(key_parts)}"

            cached = await client.get(cache_key)
            if cached is not None:
                if cached == "__NULL__":
                    return None
                return json.loads(cached)

            result = await func(*args, **kwargs)

            if result is not None:
                await client.set(cache_key, json.dumps(result, ensure_ascii=False, default=str), ex=ttl)
            else:
                await client.set(cache_key, "__NULL__", ex=null_ttl)

            return result
        return wrapper
    return decorator

# 使用
@async_redis_cache(prefix="cache:user", ttl=3600)
async def get_user_async(user_id: int) -> dict | None:
    return await query_user_from_db_async(user_id)
```

### 缓存失效：手动清除

```python
def invalidate_cache(prefix: str, *args, **kwargs):
    """手动清除指定缓存"""
    key_parts = [str(a) for a in args] + [f"{k}={v}" for k, v in sorted(kwargs.items())]
    cache_key = f"{prefix}:{':'.join(key_parts)}"
    r.delete(cache_key)

# 更新用户后清除缓存
def update_user(user_id: int, data: dict):
    update_user_in_db(user_id, data)
    invalidate_cache("cache:user", user_id)
```

## 本章小结

| 知识点 | 要点 |
|--------|------|
| Cache-Aside | 读时回填，写时删除（不是更新） |
| 穿透 | 缓存空值（`__NULL__`）+ 布隆过滤器 |
| 雪崩 | TTL 加随机抖动，避免同时过期 |
| 击穿 | 互斥锁回源，只放一个请求查 DB |
| 装饰器 | `@redis_cache(prefix, ttl)` 统一封装缓存逻辑 |
| 失效 | `invalidate_cache()` 写操作后主动删除 |
