### keys.py

```python
import time


class K:
    """Redis Key 常量"""

    # 缓存
    CACHE_ARTICLE = "cache:article:{article_id}"
    CACHE_ARTICLE_LIST = "cache:articles:page:{page}"

    # 计数
    VIEWS = "views:article:{article_id}"

    # 点赞
    LIKES = "likes:article:{article_id}"

    # 排行榜
    RANK_VIEWS = "rank:views:daily:{date}"
    RANK_LIKES = "rank:likes:daily:{date}"

    # 限流
    RATE_LIMIT = "ratelimit:{ip}:{window}"

    # 锁
    LOCK_PUBLISH = "lock:publish:{user_id}"

    @staticmethod
    def cache_article(article_id: int) -> str:
        return K.CACHE_ARTICLE.format(article_id=article_id)

    @staticmethod
    def cache_article_list(page: int) -> str:
        return K.CACHE_ARTICLE_LIST.format(page=page)

    @staticmethod
    def views(article_id: int) -> str:
        return K.VIEWS.format(article_id=article_id)

    @staticmethod
    def likes(article_id: int) -> str:
        return K.LIKES.format(article_id=article_id)

    @staticmethod
    def rank_views_today() -> str:
        return K.RANK_VIEWS.format(date=time.strftime("%Y%m%d"))

    @staticmethod
    def rank_likes_today() -> str:
        return K.RANK_LIKES.format(date=time.strftime("%Y%m%d"))
```

### redis_client.py

```python
from contextlib import asynccontextmanager
from fastapi import FastAPI
import redis.asyncio as aioredis
from config import settings


@asynccontextmanager
async def lifespan(app: FastAPI):
    pool = aioredis.ConnectionPool.from_url(
        settings.redis_url,
        max_connections=settings.redis_max_connections,
        decode_responses=True,
        socket_timeout=2.0,
        socket_connect_timeout=1.0,
    )
    app.state.redis = aioredis.Redis(connection_pool=pool)
    print(f"✅ Redis 连接池已创建 (max={settings.redis_max_connections})")
    yield
    await app.state.redis.aclose()
    await pool.disconnect()
    print("✅ Redis 连接池已释放")
```

## 10.2 实现缓存层 + 限流 + 排行榜

### services/cache.py — 通用缓存服务

```python
import json
import logging
import redis.asyncio as aioredis

logger = logging.getLogger(__name__)


class CacheService:
    """通用缓存服务（带优雅降级）"""

    def __init__(self, redis: aioredis.Redis):
        self.redis = redis

    async def get(self, key: str) -> dict | list | None:
        try:
            data = await self.redis.get(key)
            if data is None or data == "__NULL__":
                return None
            return json.loads(data)
        except Exception as e:
            logger.warning(f"缓存读取失败: {key} - {e}")
            return None

    async def set(self, key: str, value, ttl: int = 3600):
        try:
            await self.redis.set(
                key,
                json.dumps(value, ensure_ascii=False, default=str),
                ex=ttl,
            )
        except Exception as e:
            logger.warning(f"缓存写入失败: {key} - {e}")

    async def set_null(self, key: str, ttl: int = 300):
        """缓存空值（防穿透）"""
        try:
            await self.redis.set(key, "__NULL__", ex=ttl)
        except Exception:
            pass

    async def delete(self, key: str):
        try:
            await self.redis.delete(key)
        except Exception as e:
            logger.warning(f"缓存删除失败: {key} - {e}")
```

### services/article.py — 文章业务服务

```python
import json
import uuid
import redis.asyncio as aioredis
from keys import K
from config import settings
from services.cache import CacheService


# 模拟数据库
FAKE_DB: dict[int, dict] = {
    1: {"id": 1, "title": "Python Redis 实战入门", "author": "张三", "content": "Redis 是一个高性能的键值数据库..."},
    2: {"id": 2, "title": "FastAPI 最佳实践", "author": "李四", "content": "FastAPI 是一个现代的 Python Web 框架..."},
    3: {"id": 3, "title": "分布式系统设计", "author": "王五", "content": "分布式系统面临的核心挑战..."},
}
_next_id = 4


class ArticleService:
    """文章服务：整合缓存、计数、排行榜、锁"""

    def __init__(self, redis: aioredis.Redis):
        self.redis = redis
        self.cache = CacheService(redis)

    # ── 查询（Cache-Aside）──

    async def get_article(self, article_id: int) -> dict | None:
        """获取文章（缓存优先）"""
        cache_key = K.cache_article(article_id)

        # ① 查缓存
        cached = await self.cache.get(cache_key)
        if cached is not None:
            return cached

        # ② 查数据库
        article = FAKE_DB.get(article_id)
        if article is None:
            await self.cache.set_null(cache_key)
            return None

        # ③ 附加动态数据（阅读量、点赞数）
        article_with_stats = {
            **article,
            "views": int(await self.redis.get(K.views(article_id)) or 0),
            "likes": await self.redis.scard(K.likes(article_id)),
        }

        # ④ 回写缓存
        await self.cache.set(cache_key, article_with_stats, ttl=settings.cache_article_ttl)
        return article_with_stats

    async def list_articles(self, page: int = 1, size: int = 10) -> list[dict]:
        """文章列表（缓存优先）"""
        cache_key = K.cache_article_list(page)

        cached = await self.cache.get(cache_key)
        if cached is not None:
            return cached

        # 模拟分页查询
        all_ids = sorted(FAKE_DB.keys(), reverse=True)
        start = (page - 1) * size
        page_ids = all_ids[start:start + size]

        articles = []
        for aid in page_ids:
            article = FAKE_DB[aid].copy()
            article["views"] = int(await self.redis.get(K.views(aid)) or 0)
            article["likes"] = await self.redis.scard(K.likes(aid))
            articles.append(article)

        await self.cache.set(cache_key, articles, ttl=settings.cache_article_list_ttl)
        return articles

    # ── 阅读量（INCR + 排行榜）──

    async def record_view(self, article_id: int):
        """记录阅读量"""
        pipe = self.redis.pipeline()
        # 文章阅读量 +1
        pipe.incr(K.views(article_id))
        # 今日排行榜 +1
        pipe.zincrby(K.rank_views_today(), 1, str(article_id))
        pipe.expire(K.rank_views_today(), settings.ranking_ttl)
        await pipe.execute()

        # 阅读量变了，删除文章缓存
        await self.cache.delete(K.cache_article(article_id))

    # ── 点赞（Set 防重复）──

    async def toggle_like(self, article_id: int, user_id: int) -> dict:
        """点赞/取消点赞"""
        key = K.likes(article_id)

        if await self.redis.sismember(key, str(user_id)):
            # 已点赞 → 取消
            await self.redis.srem(key, str(user_id))
            await self.redis.zincrby(K.rank_likes_today(), -1, str(article_id))
            action = "unliked"
        else:
            # 未点赞 → 点赞
            await self.redis.sadd(key, str(user_id))
            await self.redis.zincrby(K.rank_likes_today(), 1, str(article_id))
            await self.redis.expire(K.rank_likes_today(), settings.ranking_ttl)
            action = "liked"

        # 删除缓存
        await self.cache.delete(K.cache_article(article_id))

        likes_count = await self.redis.scard(key)
        return {"action": action, "likes": likes_count}

    # ── 发布（分布式锁防重复提交）──

    async def publish_article(self, user_id: int, title: str, content: str) -> dict:
        """发布文章（分布式锁防重复提交）"""
        global _next_id
        lock_key = K.LOCK_PUBLISH.format(user_id=user_id)
        lock_value = str(uuid.uuid4())

        # 获取锁（5 秒内同一用户只能提交一次）
        acquired = await self.redis.set(lock_key, lock_value, nx=True, ex=5)
        if not acquired:
            return {"error": "请勿重复提交，请等待 5 秒"}

        try:
            article = {
                "id": _next_id,
                "title": title,
                "author": f"user_{user_id}",
                "content": content,
            }
            FAKE_DB[_next_id] = article
            _next_id += 1

            # 清除列表缓存
            for page in range(1, 4):
                await self.cache.delete(K.cache_article_list(page))

            return {"message": "发布成功", "article": article}
        finally:
            # 原子释放锁
            await self.redis.eval(
                "if redis.call('GET',KEYS[1])==ARGV[1] then return redis.call('DEL',KEYS[1]) else return 0 end",
                1, lock_key, lock_value,
            )
```

### services/ranking.py — 排行榜服务

```python
import redis.asyncio as aioredis
from keys import K


class RankingService:
    """排行榜服务"""

    def __init__(self, redis: aioredis.Redis):
        self.redis = redis

    async def get_top_viewed(self, top_n: int = 10) -> list[dict]:
        """今日阅读量 Top N"""
        results = await self.redis.zrevrange(
            K.rank_views_today(), 0, top_n - 1, withscores=True
        )
        return [
            {"rank": i + 1, "article_id": int(aid), "views": int(score)}
            for i, (aid, score) in enumerate(results)
        ]

    async def get_top_liked(self, top_n: int = 10) -> list[dict]:
        """今日点赞数 Top N"""
        results = await self.redis.zrevrange(
            K.rank_likes_today(), 0, top_n - 1, withscores=True
        )
        return [
            {"rank": i + 1, "article_id": int(aid), "likes": int(score)}
            for i, (aid, score) in enumerate(results)
        ]

    async def get_article_rank(self, article_id: int) -> dict:
        """查询某篇文章的排名"""
        view_rank = await self.redis.zrevrank(K.rank_views_today(), str(article_id))
        like_rank = await self.redis.zrevrank(K.rank_likes_today(), str(article_id))
        return {
            "article_id": article_id,
            "view_rank": (view_rank + 1) if view_rank is not None else None,
            "like_rank": (like_rank + 1) if like_rank is not None else None,
        }
```

### middleware.py — 限流中间件

```python
import time
from fastapi import Request
from fastapi.responses import JSONResponse
from config import settings


async def rate_limit_middleware(request: Request, call_next):
    """全局 API 限流"""
    redis = request.app.state.redis
    client_ip = request.client.host
    window = 60
    max_requests = settings.rate_limit_per_minute

    window_id = int(time.time() // window)
    rate_key = f"ratelimit:{client_ip}:{window_id}"

    try:
        current = await redis.incr(rate_key)
        if current == 1:
            await redis.expire(rate_key, window + 1)
    except Exception:
        # Redis 不可用时放行
        return await call_next(request)

    if current > max_requests:
        return JSONResponse(
            status_code=429,
            content={"detail": "请求过于频繁，请稍后再试"},
            headers={
                "X-RateLimit-Limit": str(max_requests),
                "X-RateLimit-Remaining": "0",
                "Retry-After": str(window - int(time.time() % window)),
            },
        )

    response = await call_next(request)
    response.headers["X-RateLimit-Limit"] = str(max_requests)
    response.headers["X-RateLimit-Remaining"] = str(max(0, max_requests - current))
    return response
```

### main.py — 完整入口

```python
from fastapi import FastAPI, Depends
import redis.asyncio as aioredis

from redis_client import lifespan
from dependencies import get_redis
from middleware import rate_limit_middleware
from services.article import ArticleService
from services.ranking import RankingService
from keys import K

app = FastAPI(title="Article API", lifespan=lifespan)
app.middleware("http")(rate_limit_middleware)


# ── 依赖 ──

async def get_article_service(redis: aioredis.Redis = Depends(get_redis)) -> ArticleService:
    return ArticleService(redis)

async def get_ranking_service(redis: aioredis.Redis = Depends(get_redis)) -> RankingService:
    return RankingService(redis)


# ── 文章接口 ──

@app.get("/articles")
async def list_articles(page: int = 1, svc: ArticleService = Depends(get_article_service)):
    articles = await svc.list_articles(page=page)
    return {"page": page, "articles": articles}


@app.get("/articles/{article_id}")
async def get_article(article_id: int, svc: ArticleService = Depends(get_article_service)):
    article = await svc.get_article(article_id)
    if not article:
        return {"error": "文章不存在"}, 404

    # 记录阅读量
    await svc.record_view(article_id)
    return article


@app.post("/articles")
async def publish_article(
    title: str,
    content: str,
    user_id: int = 1,
    svc: ArticleService = Depends(get_article_service),
):
    return await svc.publish_article(user_id, title, content)


@app.post("/articles/{article_id}/like")
async def toggle_like(
    article_id: int,
    user_id: int = 1,
    svc: ArticleService = Depends(get_article_service),
):
    return await svc.toggle_like(article_id, user_id)


# ── 排行榜接口 ──

@app.get("/ranking/views")
async def top_views(top: int = 10, svc: RankingService = Depends(get_ranking_service)):
    return {"ranking": await svc.get_top_viewed(top)}


@app.get("/ranking/likes")
async def top_likes(top: int = 10, svc: RankingService = Depends(get_ranking_service)):
    return {"ranking": await svc.get_top_liked(top)}


@app.get("/ranking/article/{article_id}")
async def article_rank(article_id: int, svc: RankingService = Depends(get_ranking_service)):
    return await svc.get_article_rank(article_id)


# ── 健康检查 ──

@app.get("/health")
async def health(redis: aioredis.Redis = Depends(get_redis)):
    try:
        info = await redis.info(section="memory")
        stats = await redis.info(section="stats")
        hits = stats.get("keyspace_hits", 0)
        misses = stats.get("keyspace_misses", 0)
        total = hits + misses
        hit_rate = f"{hits / total * 100:.1f}%" if total > 0 else "N/A"

        return {
            "status": "healthy",
            "redis_memory_mb": round(info.get("used_memory", 0) / 1024 / 1024, 2),
            "cache_hit_rate": hit_rate,
        }
    except Exception as e:
        return {"status": "degraded", "error": str(e)}
```

### 性能调优清单

| 优化点 | 具体措施 |
|--------|----------|
| **连接池** | 根据并发量调整 `max_connections`，异步场景 20-30 够用 |
| **Pipeline** | 多个无关联的 Redis 操作用 Pipeline 合并（如 `record_view`） |
| **TTL** | 根据数据变化频率设置合理 TTL，不要过长占内存 |
| **序列化** | 数据量大时从 JSON 换 MessagePack |
| **Key 设计** | 避免大 Key（单个 Value > 1MB），拆分成多个小 Key |
| **内存** | 设置 `maxmemory` + `maxmemory-policy`（推荐 `allkeys-lru`） |
| **监控** | 定期检查命中率，< 80% 说明缓存策略有问题 |

### Redis 内存策略配置

```bash
# redis.conf 或启动参数
maxmemory 256mb
maxmemory-policy allkeys-lru
```

| 策略 | 说明 | 适合 |
|------|------|------|
| `noeviction` | 内存满了报错（默认） | 不适合缓存 |
| `allkeys-lru` | 淘汰最近最少使用的 key | **缓存首选** ✅ |
| `volatile-lru` | 只淘汰有 TTL 的 key | 混合存储 |
| `allkeys-random` | 随机淘汰 | 无差别数据 |
